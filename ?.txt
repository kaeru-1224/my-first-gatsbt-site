  return <h1> hello!</h1>
          <p>this is first page</p>;
};

const Homepage = () => {
  return (
    <div>
      <h1> hello!</h1>
      <p>this is first page</p>
    </div>
  );
};
그런데 이렇게 작성을했을때  div가 연속적으로 작성되어 보기 어려울경우에는  fragment를 이용해주면되는데 
const Homepage = () => {
  return (
    <>
      <h1> hello!</h1>
      <p>this is first page</p>
    </>
  );
};
